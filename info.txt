def extract_ni(text: str) -> dict:
    t = norm(text)

    if not is_nota_ingreso_text(t):
        return {"numero": None, "clave": None}

    patrones = [
        r"NOTA\s+DE\s+INGRESO\s*(?:N[°º*?])?\s*:?\s*(\d{3,10})\b",
        r"\bNI\s*[:\-]\s*(\d{3,10})\b",
    ]

    for patron in patrones:
        m = re.search(patron, t, re.I)
        if m:
            numero = m.group(1).zfill(6)
            return {
                "numero": numero,
                "clave": f"NI|{numero}",
            }

    return {"numero": None, "clave": None}
