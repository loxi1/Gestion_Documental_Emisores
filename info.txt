import argparse
import re
import shutil
from pathlib import Path

from core.db import get_cursor

BASE_ENTRADA = Path("data/entrada")
BASE_TRABAJO = Path("data/trabajo")


def clean_filename(name: str) -> str:
    name = Path(name).stem.upper()

    replacements = {
        "Á": "A",
        "É": "E",
        "Í": "I",
        "Ó": "O",
        "Ú": "U",
        "Ñ": "N",
    }

    for old, new in replacements.items():
        name = name.replace(old, new)

    name = re.sub(r"[^A-Z0-9]+", "_", name)
    name = re.sub(r"_+", "_", name)

    return name.strip("_")


def obtener_ultimo_asiento(year: int, cliente: str, month: int) -> int:
    with get_cursor() as (_, cur):
        cur.execute("""
            SELECT MAX(
                CAST(
                    SUBSTRING(asiento_contable FROM '04-(\d+)')
                    AS INTEGER
                )
            ) AS maximo
            FROM (
                SELECT asiento_contable
                FROM documentos_agrupados
                WHERE cliente_abreviatura = %s
                  AND anio = %s
                  AND mes = %s

                UNION ALL

                SELECT asiento_contable
                FROM documentos_paginas
                WHERE cliente_abreviatura = %s
                  AND anio = %s
                  AND mes = %s

                UNION ALL

                SELECT asiento_contable
                FROM documentos_extraidos de
                INNER JOIN lotes_procesamiento lp
                    ON lp.id = de.lote_id
                WHERE lp.cliente_abreviatura = %s
                  AND lp.anio = %s
                  AND lp.mes = %s
            ) t
            WHERE asiento_contable ~ '^04-[0-9]+$'
        """, (
            cliente, year, month,
            cliente, year, month,
            cliente, year, month,
        ))

        row = cur.fetchone()

    return row["maximo"] or 0


def asiento_text(numero: int) -> str:
    return f"04-{numero:04d}"


def registrar_preclasificado(
    year: int,
    cliente: str,
    month: int,
    asiento: str,
    origen: Path,
    destino: Path,
):
    with get_cursor(commit=True) as (_, cur):
        cur.execute("""
            INSERT INTO archivos_preclasificados (
                cliente_abreviatura,
                anio,
                mes,
                asiento_generado,
                nombre_original,
                nombre_generado,
                ruta_original,
                ruta_generada,
                estado
            )
            VALUES (
                %s, %s, %s, %s,
                %s, %s, %s, %s,
                'preclasificado'
            )
            ON CONFLICT DO NOTHING
        """, (
            cliente,
            year,
            month,
            asiento,
            origen.name,
            destino.name,
            str(origen),
            str(destino),
        ))


def procesar(year: int, cliente: str, month: int):
    cliente = cliente.upper()
    month_str = f"{month:02d}"

    origen_dir = (
        BASE_ENTRADA
        / str(year)
        / cliente
        / month_str
        / "sin_nombre"
    )

    pendientes_dir = (
        BASE_TRABAJO
        / str(year)
        / cliente
        / month_str
        / "pendientes"
    )

    pendientes_dir.mkdir(parents=True, exist_ok=True)

    if not origen_dir.exists():
        print(f"[ERROR] No existe: {origen_dir}")
        return

    pdfs = sorted(origen_dir.glob("*.pdf"))

    if not pdfs:
        print("[INFO] No hay PDFs.")
        return

    ultimo = obtener_ultimo_asiento(year, cliente, month)

    print(f"Último asiento detectado: {asiento_text(ultimo)}")
    print(f"PDFs sin nombre: {len(pdfs)}")

    procesados = 0

    for pdf in pdfs:
        ultimo += 1

        asiento = asiento_text(ultimo)

        limpio = clean_filename(pdf.name)

        nuevo_nombre = (
            f"{asiento} "
            f"{cliente} "
            f"SIN_NOMBRE "
            f"{limpio}.pdf"
        )

        destino = pendientes_dir / nuevo_nombre

        i = 2

        while destino.exists():
            nuevo_nombre = (
                f"{asiento} "
                f"{cliente} "
                f"SIN_NOMBRE "
                f"{limpio}_{i}.pdf"
            )

            destino = pendientes_dir / nuevo_nombre
            i += 1

        shutil.copy2(pdf, destino)

        registrar_preclasificado(
            year=year,
            cliente=cliente,
            month=month,
            asiento=asiento,
            origen=pdf,
            destino=destino,
        )

        procesados += 1

        print(f"[PRE] {pdf.name}")
        print(f"      -> {nuevo_nombre}")

    print()
    print(f"Preclasificados: {procesados}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()

    parser.add_argument("--year", type=int, required=True)
    parser.add_argument("--cliente", required=True)
    parser.add_argument("--month", type=int, required=True)

    args = parser.parse_args()

    procesar(args.year, args.cliente, args.month)

CREATE TABLE IF NOT EXISTS archivos_preclasificados (
    id SERIAL PRIMARY KEY,
    cliente_abreviatura VARCHAR(20) NOT NULL,
    anio INT NOT NULL,
    mes INT NOT NULL,
    asiento_generado VARCHAR(20) NOT NULL,
    nombre_original TEXT NOT NULL,
    nombre_generado TEXT NOT NULL,
    ruta_original TEXT NOT NULL,
    ruta_generada TEXT NOT NULL,
    estado VARCHAR(30) DEFAULT 'preclasificado',
    creado_en TIMESTAMP DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_archivos_preclasificados_lote
ON archivos_preclasificados (
    cliente_abreviatura,
    anio,
    mes
);
