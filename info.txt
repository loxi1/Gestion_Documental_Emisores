4. Actualizar archivos_preclasificados
SQL
WITH base AS (
    SELECT COALESCE(
        MAX(CAST(SUBSTRING(asiento_contable FROM '04-(\d+)') AS INTEGER)),
        0
    ) AS max_num
    FROM documentos_agrupados
    WHERE cliente_abreviatura = 'BBTEC'
      AND anio = 2026
      AND mes = 4
      AND id < 1832
),
ordenados AS (
    SELECT
        ap.id,
        '04-' || LPAD(
            (base.max_num + ROW_NUMBER() OVER (ORDER BY ap.id))::TEXT,
            4,
            '0'
        ) AS asiento_nuevo
    FROM archivos_preclasificados ap, base
    WHERE ap.cliente_abreviatura = 'BBTEC'
      AND ap.anio = 2026
      AND ap.mes = 4
)
UPDATE archivos_preclasificados ap
SET asiento_generado = o.asiento_nuevo,
    actualizado_en = NOW()
FROM ordenados o
WHERE ap.id = o.id;
5. Luego marca páginas ya agrupadas
SQL
UPDATE documentos_paginas dp
SET estado = 'agrupado'
WHERE dp.cliente_abreviatura = 'BBTEC'
  AND dp.anio = 2026
  AND dp.mes = 4
  AND dp.estado = 'clasificado'
  AND EXISTS (
      SELECT 1
      FROM documentos_agrupados da
      WHERE da.cliente_abreviatura = dp.cliente_abreviatura
        AND da.anio = dp.anio
        AND da.mes = dp.mes
        AND da.archivo_fuente = dp.archivo_fuente
        AND da.clave_documental = dp.clave_documental
        AND da.origen = 'fragmentos'
  );
