import argparse
from pathlib import Path

from core.db import get_cursor

BASE_TRABAJO = Path("data/trabajo")
BASE_SALIDA = Path("data/salida")
BASE_TMP = Path("storage/tmp/pages")


def count_files(path: Path) -> int:
    if not path.exists():
        return 0
    return sum(1 for p in path.rglob("*") if p.is_file())


def print_section(title: str):
    print("\n" + "=" * 100)
    print(title)
    print("=" * 100)


def scalar(cur, sql, params):
    cur.execute(sql, params)
    row = cur.fetchone()
    return list(row.values())[0] if row else 0


def reporte(year: int, cliente: str, month: int):
    cliente = cliente.upper()
    month_str = f"{month:02d}"

    base_trabajo = BASE_TRABAJO / str(year) / cliente / month_str
    base_salida = BASE_SALIDA / str(year) / cliente / month_str
    base_tmp = BASE_TMP / str(year) / cliente / month_str

    pendientes_dir = base_trabajo / "pendientes"
    con_oc_dir = base_salida / "con_oc"
    sin_oc_dir = base_salida / "sin_oc"
    revision_dir = base_salida / "revision"
    provisional_dir = base_salida / "provisional"

    print_section(f"REPORTE INTEGRIDAD LOTE {year}/{cliente}/{month_str}")

    with get_cursor() as (_, cur):
        agrupados_total = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
        """, (cliente, year, month))

        agrupados_con_oc = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND estado='distribuido_con_oc'
        """, (cliente, year, month))

        agrupados_sin_oc = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND estado='distribuido_sin_oc'
        """, (cliente, year, month))

        agrupados_revision = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND estado='revision'
        """, (cliente, year, month))

        paginas_total = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_paginas
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
        """, (cliente, year, month))

        paginas_no_clasificadas = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_paginas
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND estado <> 'clasificado'
        """, (cliente, year, month))

        paginas_sin_clave = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_paginas
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND clave_documental IS NULL
        """, (cliente, year, month))

        qr_pendientes = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_paginas
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
              AND requiere_qr = TRUE
              AND qr_procesado = FALSE
        """, (cliente, year, month))

        extraidos_revision = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_extraidos de
            INNER JOIN lotes_procesamiento lp ON lp.id = de.lote_id
            WHERE lp.cliente_abreviatura=%s
              AND lp.anio=%s
              AND lp.mes=%s
              AND (
                    de.estado='revision'
                    OR de.requiere_revision=TRUE
                  )
        """, (cliente, year, month))

        extraidos_total = scalar(cur, """
            SELECT COUNT(*)
            FROM documentos_extraidos de
            INNER JOIN lotes_procesamiento lp ON lp.id = de.lote_id
            WHERE lp.cliente_abreviatura=%s
              AND lp.anio=%s
              AND lp.mes=%s
        """, (cliente, year, month))

        cur.execute("""
            SELECT estado, COUNT(*) total
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
            GROUP BY estado
            ORDER BY estado
        """, (cliente, year, month))
        estados_agrupados = cur.fetchall()

        cur.execute("""
            SELECT estado, COUNT(*) total
            FROM documentos_paginas
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
            GROUP BY estado
            ORDER BY estado
        """, (cliente, year, month))
        estados_paginas = cur.fetchall()

        cur.execute("""
            SELECT tipo_documental, COUNT(*) total
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s AND anio=%s AND mes=%s
            GROUP BY tipo_documental
            ORDER BY tipo_documental
        """, (cliente, year, month))
        tipos_agrupados = cur.fetchall()

        cur.execute("""
            SELECT clave_documental, COUNT(*) total
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s
              AND anio=%s
              AND mes=%s
              AND tipo_documental <> 'OTRO'
            GROUP BY clave_documental
            HAVING COUNT(*) > 1
            ORDER BY total DESC, clave_documental
        """, (cliente, year, month))
        duplicados = cur.fetchall()

        cur.execute("""
            SELECT id, asiento_contable, tipo_documental, estado, nombre_archivo, ruta_final
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s
              AND anio=%s
              AND mes=%s
              AND estado LIKE 'distribuido%%'
              AND (
                    ruta_final IS NULL
                    OR ruta_final = ''
                  )
            ORDER BY asiento_contable, id
        """, (cliente, year, month))
        sin_ruta_final = cur.fetchall()

        cur.execute("""
            SELECT id, asiento_contable, tipo_documental, estado, nombre_archivo, ruta_final
            FROM documentos_agrupados
            WHERE cliente_abreviatura=%s
              AND anio=%s
              AND mes=%s
              AND estado NOT LIKE 'distribuido%%'
            ORDER BY asiento_contable, id
        """, (cliente, year, month))
        no_distribuidos = cur.fetchall()

    fs_con_oc = count_files(con_oc_dir)
    fs_sin_oc = count_files(sin_oc_dir)
    fs_revision = count_files(revision_dir)
    fs_provisional = count_files(provisional_dir)
    fs_pendientes = count_files(pendientes_dir)
    fs_tmp_pages = count_files(base_tmp)

    print_section("RESUMEN DB")
    print(f"documentos_extraidos total      : {extraidos_total}")
    print(f"documentos_extraidos revision   : {extraidos_revision}")
    print(f"documentos_paginas total        : {paginas_total}")
    print(f"paginas no clasificadas         : {paginas_no_clasificadas}")
    print(f"paginas sin clave               : {paginas_sin_clave}")
    print(f"QR pendientes                   : {qr_pendientes}")
    print(f"documentos_agrupados total      : {agrupados_total}")
    print(f"agrupados con OC/OS             : {agrupados_con_oc}")
    print(f"agrupados sin OC/OS             : {agrupados_sin_oc}")
    print(f"agrupados revision              : {agrupados_revision}")

    print_section("RESUMEN FILESYSTEM")
    print(f"con_oc físicos                  : {fs_con_oc}")
    print(f"sin_oc físicos                  : {fs_sin_oc}")
    print(f"revision físicos                : {fs_revision}")
    print(f"provisional físicos             : {fs_provisional}")
    print(f"pendientes físicos              : {fs_pendientes}")
    print(f"tmp pages físicos               : {fs_tmp_pages}")

    print_section("VALIDACIÓN PRINCIPAL")
    total_distribuido_db = agrupados_con_oc + agrupados_sin_oc
    total_final_fs = fs_con_oc + fs_sin_oc

    print(f"DB distribuido                  : {total_distribuido_db}")
    print(f"FS final con_oc + sin_oc        : {total_final_fs}")

    if total_distribuido_db == total_final_fs:
        print("[OK] Conteo DB distribuido coincide con archivos finales.")
    else:
        print("[ALERTA] Conteo DB distribuido NO coincide con archivos finales.")

    if agrupados_revision == 0 and extraidos_revision == 0 and qr_pendientes == 0 and paginas_no_clasificadas == 0:
        print("[OK] No hay pendientes críticos.")
    else:
        print("[ALERTA] Hay pendientes por revisar.")

    print_section("ESTADOS documentos_agrupados")
    for row in estados_agrupados:
        print(f"{row['estado']}: {row['total']}")

    print_section("ESTADOS documentos_paginas")
    for row in estados_paginas:
        print(f"{row['estado']}: {row['total']}")

    print_section("TIPOS documentos_agrupados")
    for row in tipos_agrupados:
        print(f"{row['tipo_documental']}: {row['total']}")

    print_section("DUPLICADOS NO OTRO")
    if not duplicados:
        print("[OK] No hay claves duplicadas no-OTRO.")
    else:
        for row in duplicados:
            print(f"[DUP] {row['clave_documental']} -> {row['total']}")

    print_section("DISTRIBUIDOS SIN RUTA_FINAL")
    if not sin_ruta_final:
        print("[OK] Todos los distribuidos tienen ruta_final.")
    else:
        for row in sin_ruta_final:
            print(f"[SIN_RUTA] ID={row['id']} {row['asiento_contable']} {row['tipo_documental']} {row['nombre_archivo']}")

    print_section("NO DISTRIBUIDOS")
    if not no_distribuidos:
        print("[OK] No hay agrupados pendientes de distribución.")
    else:
        for row in no_distribuidos:
            print(f"[PENDIENTE] ID={row['id']} {row['asiento_contable']} {row['tipo_documental']} estado={row['estado']}")

    print_section("RESULTADO")
    errores = []

    if total_distribuido_db != total_final_fs:
        errores.append("Diferencia DB vs filesystem")

    if extraidos_revision > 0:
        errores.append("Hay extraídos en revisión")

    if qr_pendientes > 0:
        errores.append("Hay QR pendientes")

    if paginas_no_clasificadas > 0:
        errores.append("Hay páginas no clasificadas")

    if paginas_sin_clave > 0:
        errores.append("Hay páginas sin clave")

    if agrupados_revision > 0:
        errores.append("Hay agrupados en revisión")

    if no_distribuidos:
        errores.append("Hay documentos agrupados no distribuidos")

    if sin_ruta_final:
        errores.append("Hay distribuidos sin ruta_final")

    if duplicados:
        errores.append("Hay claves duplicadas no-OTRO")

    if errores:
        print("[LOTE CON OBSERVACIONES]")
        for e in errores:
            print(f"- {e}")
    else:
        print("[LOTE OK] Todo el lote parece procesado correctamente.")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--year", type=int, required=True)
    parser.add_argument("--cliente", required=True)
    parser.add_argument("--month", type=int, required=True)

    args = parser.parse_args()

    reporte(args.year, args.cliente, args.month)
