def extract_ni(text: str) -> dict:
    t = norm(text)

    if not is_nota_ingreso_text(t):
        return {"numero": None, "clave": None}

    m = re.search(
        r"NOTA\s+DE\s+INGRESO\s+0*(\d{3,10})",
        t,
        re.I | re.S
    )

    if not m:
        return {"numero": None, "clave": None}

    numero = m.group(1).zfill(10)

    return {
        "numero": numero,
        "clave": f"NI|{numero}",
    }
