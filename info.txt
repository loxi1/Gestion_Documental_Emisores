Tu tabla documentos_agrupados no tiene la columna nombre_archivo.
1. Arregla la tabla
Ejecuta:
SQL
ALTER TABLE documentos_agrupados
ADD COLUMN IF NOT EXISTS asiento_contable VARCHAR(20),
ADD COLUMN IF NOT EXISTS clave_documental TEXT,
ADD COLUMN IF NOT EXISTS tipo_documental VARCHAR(50),
ADD COLUMN IF NOT EXISTS nombre_archivo TEXT,
ADD COLUMN IF NOT EXISTS ruta_archivo TEXT,
ADD COLUMN IF NOT EXISTS paginas TEXT,
ADD COLUMN IF NOT EXISTS estado VARCHAR(50) DEFAULT 'agrupado',
ADD COLUMN IF NOT EXISTS creado_en TIMESTAMP DEFAULT now();
Luego:
SQL
TRUNCATE documentos_agrupados RESTART IDENTITY CASCADE;
