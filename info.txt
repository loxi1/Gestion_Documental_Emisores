def extract_nota_credito_from_text(text: str) -> dict:
    t = norm(text)

    ruc = None
    serie = None
    numero = None

    m = re.search(
        r"NOTA\s+DE\s+CR[ÉE]DITO\s+ELECTR[ÓO]NICA[\s\S]{0,120}?RUC\s*:?\s*((10|20)\d{9})",
        t,
        re.I,
    )

    if not m:
        m = re.search(
            r"R\.?\s*U\.?\s*C\.?\s*(?:N[°º])?\s*:?\s*((10|20)\d{9})",
            t,
            re.I,
        )

    if m:
        ruc = m.group(1)

    patterns = [
        r"NOTA\s+DE\s+CR[ÉE]DITO\s+ELECTR[ÓO]NICA[\s\S]{0,120}?([EF][A-Z0-9]{3})\s*[-–—]\s*0*(\d{1,12})",
        r"\b([EF][A-Z0-9]{3})\s*[-–—]\s*0*(\d{1,12})\b",
    ]

    for pattern in patterns:
        m = re.search(pattern, t, re.I)
        if m:
            serie = normalize_serie(m.group(1))
            numero = m.group(2).lstrip("0") or "0"
            break

    clave = f"NC|{ruc}|{serie}|{numero}" if ruc and serie and numero else None

    return {
        "serie": serie,
        "numero": numero,
        "ruc": ruc,
        "clave": clave,
    }
