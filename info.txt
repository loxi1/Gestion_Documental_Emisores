este ultimo me sale asi porque creo que ejecute este:
UPDATE documentos_paginas dp
SET estado = 'agrupado'
WHERE dp.cliente_abreviatura = 'BBTEC'
  AND dp.anio = 2026
  AND dp.mes = 4
  AND dp.estado = 'clasificado'
  AND EXISTS (
      SELECT 1
      FROM documentos_agrupados da
      WHERE da.cliente_abreviatura = dp.cliente_abreviatura
        AND da.anio = dp.anio
        AND da.mes = dp.mes
        AND da.asiento_contable = dp.asiento_contable
        AND da.archivo_fuente = dp.archivo_fuente
        AND da.clave_documental = dp.clave_documental
        AND da.origen = 'fragmentos'
  );
