import argparse
from pathlib import Path

import fitz

from core.db import get_cursor


BASE_TRABAJO = Path("data/trabajo")
BASE_SALIDA = Path("data/salida")


def mismo_documento(prev, actual) -> bool:
    if prev["archivo_fuente"] != actual["archivo_fuente"]:
        return False

    if actual.get("es_reverso"):
        return True

    if prev["tipo_detectado_basico"] != actual["tipo_detectado_basico"]:
        return False

    if not prev.get("clave_documental") or not actual.get("clave_documental"):
        return False

    return prev["clave_documental"] == actual["clave_documental"]


def crear_grupos(rows):
    grupos = []
    actual = []

    for row in rows:
        if not actual:
            actual.append(row)
            continue

        prev = actual[-1]

        if row["pagina"] == prev["pagina"] + 1 and mismo_documento(prev, row):
            actual.append(row)
        else:
            grupos.append(actual)
            actual = [row]

    if actual:
        grupos.append(actual)

    return grupos


def nombre_provisional(grupo):
    first = grupo[0]
    last = grupo[-1]

    asiento = first["asiento_contable"]
    tipo = first["tipo_detectado_basico"]
    clave = first.get("clave_documental") or ""

    p_ini = first["pagina"]
    p_fin = last["pagina"]

    if clave.startswith("FACTURA|"):
        _, ruc, serie, numero = clave.split("|")
        return f"{asiento} FACTURA {serie} {numero} {ruc} P{p_ini:03d}-P{p_fin:03d}.pdf"

    if clave.startswith("GUIA|"):
        _, ruc, serie, numero = clave.split("|")
        return f"{asiento} GUIA_REMISION {serie} {numero} {ruc} P{p_ini:03d}-P{p_fin:03d}.pdf"

    if clave.startswith("OS|"):
        numero = clave.split("|")[1]
        return f"{asiento} OS {numero} P{p_ini:03d}-P{p_fin:03d}.pdf"

    if clave.startswith("OC|"):
        numero = clave.split("|")[1]
        return f"{asiento} OC {numero} P{p_ini:03d}-P{p_fin:03d}.pdf"

    if clave.startswith("NI|"):
        numero = clave.split("|")[1]
        return f"{asiento} NOTA_INGRESO {numero} P{p_ini:03d}-P{p_fin:03d}.pdf"

    tipo_nombre = (tipo or "OTRO").upper()
    return f"{asiento} {tipo_nombre} P{p_ini:03d}-P{p_fin:03d}.pdf"


def exportar_paginas(pdf_origen: Path, paginas: list[int], destino: Path):
    destino.parent.mkdir(parents=True, exist_ok=True)

    src = fitz.open(pdf_origen)
    dst = fitz.open()

    for pagina in paginas:
        dst.insert_pdf(src, from_page=pagina - 1, to_page=pagina - 1)

    dst.save(destino)

    dst.close()
    src.close()


def registrar_agrupado(grupo, destino: Path, nombre: str):
    first = grupo[0]
    last = grupo[-1]
    clave = first.get("clave_documental") or ""

    orden_servicio = None
    orden_compra = None

    if clave.startswith("OS|"):
        orden_servicio = clave.split("|")[1]

    if clave.startswith("OC|"):
        orden_compra = clave.split("|")[1]

    with get_cursor(commit=True) as (_, cur):
        cur.execute(
            """
            INSERT INTO documentos_agrupados (
                asiento_contable,
                archivo_fuente,
                tipo_documental,
                clave_documental,
                serie,
                numero,
                orden_servicio,
                orden_compra,
                pagina_inicio,
                pagina_fin,
                paginas,
                nombre_provisional,
                ruta_provisional,
                estado
            )
            VALUES (%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,%s,'agrupado')
            """,
            (
                first["asiento_contable"],
                first["archivo_fuente"],
                first["tipo_detectado_basico"],
                clave,
                first.get("serie"),
                first.get("numero"),
                orden_servicio,
                orden_compra,
                first["pagina"],
                last["pagina"],
                ",".join(str(x["pagina"]) for x in grupo),
                nombre,
                str(destino),
            ),
        )


def process(year: int, cliente: str, month: int):
    pendientes = BASE_TRABAJO / str(year) / cliente / f"{month:02d}" / "pendientes"
    salida = BASE_SALIDA / str(year) / cliente / f"{month:02d}" / "provisional"

    with get_cursor() as (_, cur):
        cur.execute(
            """
            SELECT *
            FROM documentos_paginas
            WHERE estado = 'enriquecido'
               OR estado = 'detectado'
            ORDER BY archivo_fuente, pagina
            """
        )
        rows = cur.fetchall()

    por_archivo = {}

    for row in rows:
        por_archivo.setdefault(row["archivo_fuente"], []).append(row)

    total = 0

    for archivo_fuente, paginas in por_archivo.items():
        pdf_origen = pendientes / archivo_fuente

        if not pdf_origen.exists():
            print(f"[WARN] No existe fuente: {pdf_origen}")
            continue

        grupos = crear_grupos(paginas)

        for grupo in grupos:
            nombre = nombre_provisional(grupo)
            destino = salida / nombre
            paginas_pdf = [x["pagina"] for x in grupo]

            exportar_paginas(pdf_origen, paginas_pdf, destino)
            registrar_agrupado(grupo, destino, nombre)

            total += 1
            print(f"[AGRUPADO] {nombre}")

    print(f"Total agrupados: {total}")


if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--year", type=int, required=True)
    parser.add_argument("--cliente", required=True)
    parser.add_argument("--month", type=int, required=True)

    args = parser.parse_args()

    process(args.year, args.cliente, args.month)

import argparse
import re
from pathlib import Path

import fitz

from core.db import get_cursor
from core.ocr_runner import run_ocr
from core.classifier import detect_tipo_documental


BASE_TRABAJO = Path("data/trabajo")
TMP_DIR = Path("storage/tmp/pages")


def normalize(txt: str) -> str:
    txt = txt or ""
    txt = txt.replace("\x00", "").replace("\u0000", "")
    return re.sub(r"\s+", " ", txt.upper()).strip()


def limpiar_para_postgres(txt: str) -> str:
    if not txt:
        return ""
    return txt.replace("\x00", "").replace("\u0000", "")


def extract_asiento(filename: str) -> str:
    m = re.search(r"^(04-\d{4})", filename)
    return m.group(1) if m else "SIN_ASIENTO"


def extract_text(pdf_path: Path) -> str:
    text = []

    with fitz.open(pdf_path) as doc:
        for page in doc:
            text.append(page.get_text("text"))

    return "\n".join(text)


def split_page(source_pdf: Path, page_index: int, output_pdf: Path):
    output_pdf.parent.mkdir(parents=True, exist_ok=True)

    src = fitz.open(source_pdf)
    dst = fitz.open()

    dst.insert_pdf(src, from_page=page_index, to_page=page_index)
    dst.save(output_pdf)

    dst.close()
    src.close()


def is_reverso_page(text: str) -> bool:
    text = normalize(text)

    if len(text) < 80:
        return True

    if "USUARIO:" in text and len(text) < 300:
        return True

    return False


def process(year: int, cliente: str, month: int):
    pendientes = BASE_TRABAJO / str(year) / cliente / f"{month:02d}" / "pendientes"

    pdfs = sorted(pendientes.glob("*.pdf"))

    print(f"Carpeta pendientes: {pendientes}")
    print(f"PDF pendientes: {len(pdfs)}")

    for pdf in pdfs:
        asiento = extract_asiento(pdf.name)

        with fitz.open(pdf) as doc:
            total_pages = doc.page_count

        print(f"\n[PDF] {pdf.name} ({total_pages} páginas)")

        for page_index in range(total_pages):
            page_pdf = TMP_DIR / f"{pdf.stem}_P{page_index + 1}.pdf"
            page_ocr = TMP_DIR / f"{pdf.stem}_P{page_index + 1}_OCR.pdf"

            split_page(pdf, page_index, page_pdf)

            texto_pdf = extract_text(page_pdf)

            fuente_texto = "pdf_native"
            pagina_digital = False
            pagina_ocr = False

            text = texto_pdf

            if len(normalize(texto_pdf)) > 30:
                pagina_digital = True
            else:
                ok = run_ocr(page_pdf, page_ocr)

                if ok:
                    text = extract_text(page_ocr)
                    fuente_texto = "ocr"
                    pagina_ocr = True

            text_clean = limpiar_para_postgres(text)
            text_u = normalize(text_clean)

            tipo_basico = detect_tipo_documental(text_u, pdf.name)

            reverso = is_reverso_page(text_u)

            print(
                f"  Página {page_index + 1}: "
                f"tipo={tipo_basico} "
                f"fuente={fuente_texto} "
                f"digital={pagina_digital} "
                f"ocr={pagina_ocr} "
                f"reverso={reverso}"
            )

            with get_cursor(commit=True) as (_, cur):
                cur.execute(
                    """
                    INSERT INTO documentos_paginas (
                        asiento_contable,
                        archivo_fuente,
                        pagina,
                        tipo_detectado_basico,
                        texto_extraido,
                        fuente_texto,
                        pagina_digital,
                        pagina_ocr,
                        es_reverso,
                        imagen_generada,
                        estado
                    )
                    VALUES (
                        %s,%s,%s,%s,%s,%s,%s,%s,%s,TRUE,'detectado'
                    )
                    ON CONFLICT (archivo_fuente, pagina)
                    DO UPDATE SET
                        tipo_detectado_basico = EXCLUDED.tipo_detectado_basico,
                        texto_extraido = EXCLUDED.texto_extraido,
                        fuente_texto = EXCLUDED.fuente_texto,
                        pagina_digital = EXCLUDED.pagina_digital,
                        pagina_ocr = EXCLUDED.pagina_ocr,
                        es_reverso = EXCLUDED.es_reverso,
                        estado = 'detectado'
                    """,
                    (
                        asiento,
                        pdf.name,
                        page_index + 1,
                        tipo_basico,
                        text_clean[:10000],
                        fuente_texto,
                        pagina_digital,
                        pagina_ocr,
                        reverso,
                    ),
                )


if __name__ == "__main__":
    parser = argparse.ArgumentParser()

    parser.add_argument("--year", type=int, required=True)
    parser.add_argument("--cliente", required=True)
    parser.add_argument("--month", type=int, required=True)

    args = parser.parse_args()

import re

from core.db import get_cursor


def normalize(txt: str) -> str:
    return re.sub(r"\s+", " ", (txt or "").upper()).strip()


def extract_factura_data(text):
    m = re.search(r"\b(F[A-Z0-9]{2,4})[- ]?0*(\d+)\b", text)
    ruc = re.search(r"\b(10|20)\d{9}\b", text)

    if not m:
        return None

    return {
        "serie": m.group(1).upper(),
        "numero": m.group(2),
        "ruc": ruc.group(0) if ruc else None,
    }


def enrich_pages():
    with get_cursor(commit=True) as (_, cur):
        cur.execute(
            """
            SELECT *
            FROM documentos_paginas
            WHERE estado = 'detectado'
            ORDER BY archivo_fuente, pagina
            """
        )

        rows = cur.fetchall()

    for row in rows:
        text = normalize(row["texto_extraido"])

        clave = None
        serie = None
        numero = None
        ruc = None

        tipo = row["tipo_detectado_basico"]

        if tipo == "factura":
            data = extract_factura_data(text)

            if data:
                serie = data["serie"]
                numero = data["numero"]
                ruc = data["ruc"]

                clave = f"FACTURA|{ruc}|{serie}|{numero}"

        elif tipo == "guia_remision":
            pass

        elif tipo == "orden_servicio":
            pass

        elif tipo == "orden_compra":
            pass

        elif tipo == "nota_ingreso":
            pass

        with get_cursor(commit=True) as (_, cur):
            cur.execute(
                """
                UPDATE documentos_paginas
                SET
                    clave_documental = %s
                WHERE id = %s
                """,
                (
                    clave,
                    row["id"],
                ),
            )


if __name__ == "__main__":
    enrich_pages()
